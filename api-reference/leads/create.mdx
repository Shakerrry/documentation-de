---
title: "Lead erstellen"
api: "POST /user/lead"
description: "Einen neuen Lead im Autocalls-System erstellen"
---

Dieser Endpunkt ermöglicht es Ihnen, einen neuen Lead im Autocalls-System zu erstellen.

### Anfragekörper

<ParamField body="phone_number" type="string" required>
  Die Telefonnummer des Leads im E.164-Format (z. B. +1234567890)
</ParamField>

<ParamField body="campaign_id" type="integer" required>
  Die ID der Kampagne, für die der Lead erstellt werden soll
</ParamField>

<ParamField body="variables" type="object">
  Die Variablen, die an den Lead übergeben werden sollen
  
  <Expandable title="Variablen-Eigenschaften">
    <ParamField body="customer_name" type="string" default="<string>">
      Der Name des Kunden
    </ParamField>
    
    <ParamField body="email" type="string" default="<string>">
      Die E-Mail-Adresse des Kunden
    </ParamField>
  </Expandable>
</ParamField>

<ParamField body="allow_dupplicate" type="boolean">
  Ob doppelte Leads in einer Kampagne erlaubt werden sollen.
</ParamField>

<ParamField body="secondary_contacts" type="array">
  Array von sekundären Kontakt-Leads, die zusammen mit dem Haupt-Lead erstellt werden sollen
  
  <Expandable title="Eigenschaften sekundärer Kontakte">
    <ParamField body="phone_number" type="string" required>
      Die Telefonnummer des sekundären Kontakts im E.164-Format
    </ParamField>
    
    <ParamField body="variables" type="object">
      Die Variablen, die an den sekundären Kontakt übergeben werden sollen
      
      <Expandable title="Variablen-Eigenschaften">
        <ParamField body="customer_name" type="string" default="<string>">
          Der Name des sekundären Kontakts
        </ParamField>
        
        <ParamField body="email" type="string" default="<string>">
          Die E-Mail-Adresse des sekundären Kontakts
        </ParamField>
      </Expandable>
    </ParamField>
  </Expandable>
</ParamField>

### Antwort

<ResponseField name="message" type="string" default="Lead created successfully">
  Die Nachricht der Antwort
</ResponseField>

<ResponseField name="data" type="object">
  Das Lead-Datenobjekt
  
  <Expandable title="Daten-Eigenschaften">
    <ResponseField name="id" type="integer">
      Die eindeutige ID des erstellten Leads
    </ResponseField>
    
    <ResponseField name="campaign_id" type="integer">
      Die ID der Kampagne, zu der dieser Lead gehört
    </ResponseField>

    <ResponseField name="phone_number" type="string">
      Die Telefonnummer des Leads im E.164-Format
    </ResponseField>
    
    <ResponseField name="variables" type="object">
      Die dem Lead zugeordneten Variablen
      
      <Expandable title="Variablen-Eigenschaften">
        <ResponseField name="customer_name" type="string" default="<string>">
          Der Name des Kunden
        </ResponseField>
        
        <ResponseField name="email" type="string" default="<string>">
          Die E-Mail-Adresse des Kunden
        </ResponseField>
      </Expandable>
    </ResponseField>
    
    <ResponseField name="status" type="string" default="created">
      Der Status des Leads
    </ResponseField>
    
    <ResponseField name="created_at" type="string">
      Der Zeitstempel, wann der Lead erstellt wurde
    </ResponseField>
    
    <ResponseField name="updated_at" type="string">
      Der Zeitstempel, wann der Lead zuletzt aktualisiert wurde
    </ResponseField>

    <ResponseField name="campaign" type="object">
      Die Kampagneninformationen
      <Expandable title="Kampagnen-Eigenschaften">
        <ResponseField name="id" type="integer">
          Die ID der Kampagne
        </ResponseField>
        <ResponseField name="name" type="string">
          Der Name der Kampagne
        </ResponseField>
      </Expandable>
    </ResponseField>

    <ResponseField name="secondary_contacts" type="array">
      Array von sekundären Kontakt-Leads, die mit diesem Lead verknüpft sind
      <Expandable title="Eigenschaften sekundärer Kontakte">
        <ResponseField name="id" type="integer">
          Die ID des sekundären Kontakt-Leads
        </ResponseField>
        <ResponseField name="phone_number" type="string">
          Die Telefonnummer des sekundären Kontakts im E.164-Format
        </ResponseField>
        <ResponseField name="variables" type="object">
          Die dem sekundären Kontakt zugeordneten Variablen
        </ResponseField>
        <ResponseField name="status" type="string">
          Der Status des sekundären Kontakts
        </ResponseField>
        <ResponseField name="created_at" type="string">
          Das Erstellungsdatum des sekundären Kontakts
        </ResponseField>
        <ResponseField name="updated_at" type="string">
          Das Aktualisierungsdatum des sekundären Kontakts
        </ResponseField>
      </Expandable>
    </ResponseField>
  </Expandable>
</ResponseField>
